room = []

    try:
        s = df.loc[i,'房型'].strip(" ")
        if len(s)==8:
            room.append([df.loc[i,'房型'][0],
                      df.loc[i,'房型'][2],
                      df.loc[i,'房型'][4],
                      df.loc[i,'房型'][6]])

            room.append([-1,-1,-1,-1])

        room.append([-1,-1,-1,-1])
